294  // Binds parameters to each Slot so Combiner does not need to know them.
295  template <typename... Params>
296  auto bind_args(Params&&... args) const -> Bound_slot_container
297  {
298  auto bound_slots = Bound_slot_container{};
299  // Helper Function
300  auto bind_slots = [&bound_slots, &args...](auto& conn_container) {
301  for (auto& connection : conn_container) {
302  if (connection->connected() && !connection->blocked() &&
303  !connection->get_slot().expired()) {
304  auto& slot = connection->get_slot();
305  bound_slots.push_back(
306  [slot, &args...] { return slot(args...); });
307  }
308  }
309  };
310  // Bind arguments to all three types of connected Slots.
311  auto const lock = std::lock_guard{mtx_};
312  bind_slots(connections_.front);
313  for (auto& group : connections_.grouped)
314  bind_slots(group.second);
315  bind_slots(connections_.back);
316  return bound_slots;
317  }
