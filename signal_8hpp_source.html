420  {
421  auto bound_slots = Bound_slot_container{};
422  // Helper Function
423  auto bind_slots = [&bound_slots, &args...](auto& conn_container) {
424  for (auto& connection : conn_container) {
425  if (connection->connected() && !connection->blocked() &&
426  !connection->get_slot().expired()) {
427  auto& slot = connection->get_slot();
428  // you get lucky the args... are still in scope while needed
429  bound_slots.push_back(
430  [&slot, &args...] { return slot(args...); });
431  }
432  }
433  };
434  // Bind arguments to all three types of connected Slots.
435  auto const lock = Lock_t{mtx_};
436  bind_slots(connections_.front);
437  for (auto& group : connections_.grouped)
438  bind_slots(group.second);
439  bind_slots(connections_.back);
